deg_spatial_density = data.frame(miceage = c(), degree = c(), density = c())

degree = p2_degree$degree
deg_density = table(degree)[max(degree):1] %>%
  prop.table() %>%
  cumsum() %>%
  rev()
deg_spatial_density = rbind(deg_spatial_density,
